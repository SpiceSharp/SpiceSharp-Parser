<NETLIST> ::= <TITLE> <NEW-LINE> <STATEMENTS> <NETLIST-ENDING>
<NETLIST-ENDING> ::= <END> <EOF> | <END> <NEW-LINE> <EOF>
<STATEMENTS> ::= <STATEMENT> <STATEMENTS> | <NEW-LINE> <STATEMENTS> | <EMPTY> 

<STATEMENT> ::= <COMPONENT> <NEW-LINE> | <SUBCKT> <NEW-LINE> | <MODEL> <NEW-LINE> | <CONTROL> <NEW-LINE> | <COMMENT-LINE> <NEW-LINE> 
<COMPONENT> ::= <WORD> <PARAMETERS>
<COMMENT-LINE> ::= <ASTERIKS> <COMMENT> | <ASTERIKS>
<SUBCKT> ::= <DOT> "SUBCKT" <WORD> <PARAMETERS> <NEW-LINE> <STATEMENTS> <SUBCKT-ENDING>
<SUBCKT-ENDING> ::= <ENDS> <WORD>
<MODEL> ::= <DOT> "MODEL" <WORD> <PARAMETERS>
<CONTROL ::= <DOT> <WORD> <PARAMETERS>

<PARAMETERS> ::= <PARAMETER> <PARAMETERS> | <EMPTY> 
<PARAMETER> ::= <VECTOR>
			| <PARAMETER-EQUAL>
			| <PARAMETER-BRACKET>
			| <PARAMETER-SINGLE> 
			
<PARAMETER-EQUAL> ::= <PARAMETER-EQUAL-SINGLE> 
				  | <WORD> <LEFT-BRACKET> <PARAMETER-SINGLE> <RIGHT-BRACKET> <EQUAL> <PARAMTER-SINGLE>
   				  | <WORD> <LEFT-BRACKET> <PARAMETER-SINGLE> <COMMA> <PARAMETER-SINGLE> <RIGHT-BRACKET> <EQUAL> <PARAMTER-SINGLE>
	
<PARAMETER-EQUAL-SINGLE> :: = <WORD> <EQUAL> <PARAMETER-SINGLE>

<PARAMETER-BRACKET> ::= <WORD> <LEFT-BRACKET> <PARAMETER-BRACKET-CONTENT> <RIGHT-BRACKET> 
<PARAMETER-BRACKET-CONTENT> ::= <PARAMETERS>

<VECTOR> ::= <PARAMETER-SINGLE> "," <PARAMETER-SINGLE> <VECTOR-CONTINUE> 
<VECTOR-CONTINUE> ::= "," <PARAMETER-SINGLE> <VECTOR-CONTINUE> | <EMPTY>

<PARAMETER-SINGLE> ::= <WORD> | <VALUE> | <STRING> | <IDENTIFIER> | <REFERENCE> | <EXPRESSION> 
<EMPTY> ::= ""
