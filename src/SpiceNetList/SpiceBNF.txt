<START> ::= <TITLE> <STATEMENTS>
<STATEMENTS> ::= <STATEMENT> <STATEMENTS> | <EMPTY> | <NEW-LINE> <STATEMENTS> | ".END"
<STATEMENT> ::= <COMPONENT> | <SUBCKT> | <MODEL> | <CONTROL> | <COMMENT-LINE>
<COMMENT-LINE> ::= <ASTERIKS> <COMMENT> | <ASTERIKS> <NEW-LINE-OR-EOF> 
<COMPONENT> ::= <WORD> <PARAMETERS> <NEW-LINE-OR-EOF> 
<SUBCKT> ::= <DOT> "SUBCKT" <WORD> <PARAMETERS> <NEW-LINE> <STATEMENTS> <SUBCKT-ENDING>
<SUBCKT-ENDING> ::= <ENDS> <NEW-LINE-OR-EOF> | <ENDS> <WORD> <NEW-LINE-OR-EOF>
<MODEL> ::= <DOT> "MODEL" <WORD> <PARAMETERS> <NEW-LINE-OR-EOF>
<CONTROL ::= <DOT> <WORD> <PARAMETERS> <NEW-LINE-OR-EOF>
<PARAMETERS> ::= <PARAMETER> <PARAMETERS> | <EMPTY>
<PARAMETER> ::= <WORD> <EQUAL> <VALUE> 
	| <WORD> <LEFT-BRACKET> <PARAMETERS> <RIGHT-BRACKET> 
	| <PARAMETER-SINGLE>
<PARAMETER-SINGLE> ::= <WORD> | <VALUE> | <STRING> | <IDENTIFIER> | <REFERENCE> | <EXPRESSION> 
<NEW-LINE-OR-EOF> ::= <NEW-LINE> | <EOF>

<EMPTY> ::= ""
